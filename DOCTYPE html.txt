    <script>
        let conversationHistory = [];
        let currentTopic = null;
        let userLevel = 'intermediate'; // beginner, intermediate, advanced
        let questionsAsked = 0;
        let conceptsLearned = 0;
        let reflectionsMade = 0;
        const psychologyKnowledge = {
            anxiety: {
                concepts: ['GAD', 'panic disorder', 'phobias', 'social anxiety', 'PTSD', 'fight-flight-freeze'],
                examples: ['exam anxiety', 'public speaking fears', 'social situations', 'trauma responses'],
                analogies: ['alarm system', 'smoke detector sensitivity', 'protective bubble'],
                questions: [
                    "What do you think happens in your body when you feel anxious?",
                    "Can you think of a time when anxiety might actually be helpful?",
                    "How might someone's past experiences shape their anxiety responses?",
                    "What's the difference between normal worry and clinical anxiety?"
                ]
            },
            depression: {
                concepts: ['MDD', 'dysthymia', 'bipolar', 'neurotransmitters', 'cognitive triad', 'learned helplessness'],
                examples: ['seasonal depression', 'postpartum depression', 'grief vs depression'],
                analogies: ['dark cloud', 'broken filter', 'depleted battery'],
                questions: [
                    "How would you explain the difference between sadness and depression?",
                    "What role do you think thoughts play in maintaining depression?",
                    "Why might some people be more vulnerable to depression than others?",
                    "How do you think depression affects someone's view of the future?"
                ]
            },
            therapy: {
                concepts: ['CBT', 'DBT', 'psychodynamic', 'humanistic', 'therapeutic alliance', 'transference'],
                examples: ['thought records', 'exposure therapy', 'mindfulness', 'free association'],
                analogies: ['mental gym', 'detective work', 'gardening the mind'],
                questions: [
                    "What do you think makes therapy effective?",
                    "How might different therapy approaches suit different people?",
                    "What role does the relationship between therapist and client play?",
                    "Why might someone resist change even when they want to feel better?"
                ]
            },
            stress: {
                concepts: ['acute vs chronic stress', 'cortisol', 'coping mechanisms', 'resilience', 'stress-vulnerability model'],
                examples: ['work stress', 'academic pressure', 'relationship conflicts', 'major life changes'],
                analogies: ['pressure cooker', 'rubber band', 'muscle fatigue'],
                questions: [
                    "What's the difference between stress and stressors?",
                    "How do you think long-term stress affects the body and mind?",
                    "What makes some people more resilient to stress than others?",
                    "Can you think of ways stress might sometimes be beneficial?"
                ]
            },
            cognitive: {
                concepts: ['schemas', 'automatic thoughts', 'cognitive distortions', 'information processing', 'attention', 'memory bias'],
                examples: ['catastrophizing', 'confirmation bias', 'mind reading', 'selective attention'],
                analogies: ['mind as a computer', 'lens/filter', 'mental shortcut'],
                questions: [
                    "How do thought patterns influence emotions and behavior?",
                    "What is a cognitive distortion and can you give an example?",
                    "Why do people sometimes jump to conclusions?",
                    "How do attentional biases affect perception?"
                ]
            },
            assessment: {
                concepts: ['clinical interview', 'psychometrics', 'screening tools', 'validity', 'reliability', 'diagnostic criteria'],
                examples: ['Beck Depression Inventory', 'MMPI', 'structured interview'],
                analogies: ['medical checkup', 'data gathering', 'puzzle'],
                questions: [
                    "What makes a good psychological assessment?",
                    "How do clinicians decide which assessment tools to use?",
                    "Why is reliability important?",
                    "How would you balance structured vs unstructured assessment?"
                ]
            }
        };
        function addMessage(message, isUser = false) {
            const chatContainer = document.getElementById('chatContainer');
            const messageDiv = document.createElement('div');
            messageDiv.className = 'message';
            const messageClass = isUser ? 'user-message' : 'tutor-message';
            messageDiv.innerHTML = `<div class="${messageClass}">${message}</div>`;
            chatContainer.appendChild(messageDiv);
            chatContainer.scrollTop = chatContainer.scrollHeight;
            conversationHistory.push({message, isUser, timestamp: Date.now()});
        }
        function generateResponse(userInput) {
            const input = userInput.toLowerCase();
            if (input.includes('i don\'t understand') || input.includes('confused')) {
                userLevel = 'beginner';
            } else if (input.includes('research') || input.includes('studies') || input.includes('neuroscience')) {
                userLevel = 'advanced';
            }
            if (input.includes('i feel') || input.includes('my experience') || input.includes('personally')) {
                reflectionsMade++;
                document.getElementById('reflectionsMade').textContent = reflectionsMade;
            }
            if (currentTopic && psychologyKnowledge[currentTopic]) {
                const topicData = psychologyKnowledge[currentTopic];
                for (let concept of topicData.concepts) {
                    if (input.includes(concept.toLowerCase())) {
                        conceptsLearned++;
                        document.getElementById('conceptsLearned').textContent = conceptsLearned;
                        break;
                    }
                }
                if (Math.random() < 0.6) {
                    const randomQuestion = topicData.questions[Math.floor(Math.random() * topicData.questions.length)];
                    return `That's an interesting point! ${randomQuestion} Take your time to think about it - there's no rush.`;
                }
            }
            const responses = [
                `I can see you're thinking deeply about this. What led you to that conclusion? Can you walk me through your reasoning?`,
                `That's a thoughtful response! How do you think this concept might apply to real-world situations you've encountered?`,
                `Interesting perspective! What evidence or examples can you think of that might support or challenge this idea?`,
                `You're making good connections! How might someone with a different background or experience view this differently?`,
                `I notice you mentioned [concept]. Can you explain that in your own words and give me an example of how it might work?`,
                `That shows good critical thinking! What questions does this raise for you about mental health treatment?`,
                `You're on the right track! How do you think this relates to what we know about the brain and behavior?`,
                `Great insight! If you were explaining this to a friend who's not studying psychology, how would you put it?`
            ];
            return responses[Math.floor(Math.random() * responses.length)];
        }
        function sendMessage() {
            const input = document.getElementById('userInput');
            const message = input.value.trim();
            if (message === '') return;
            addMessage(message, true);
            input.value = '';
            questionsAsked++;
            document.getElementById('questionsAsked').textContent = questionsAsked;
            const progress = Math.min((questionsAsked * 10 + conceptsLearned * 15 + reflectionsMade * 20), 100);
            document.getElementById('progressFill').style.width = progress + '%';
            setTimeout(() => {
                const response = generateResponse(message);
                addMessage(response);
            }, 900 + Math.random() * 700);
        }
        function startTopic(topic) {
            currentTopic = topic;
            document.getElementById('topicFeedback').style.display = 'flex';
            const topicData = psychologyKnowledge[topic];
            const responses = {
                anxiety: `Great choice! Anxiety disorders are among the most common mental health conditions. Let's start with your current understanding - when you hear "anxiety disorder," what comes to mind? How would you distinguish between normal anxiety and something that might need clinical attention?`,
                depression: `Depression is such an important topic in mental health. Before we dive deep, I'm curious about your perspective - how would you explain to someone the difference between feeling sad and experiencing clinical depression? What factors do you think contribute to someone developing depression?`,
                therapy: `Therapeutic approaches are fascinating! There are so many different ways to help people heal and grow. What do you think makes therapy effective? Have you wondered why different approaches might work better for different people or problems?`,
                stress: `Stress is something we all experience, but it affects everyone differently. Let's explore this together - can you think of the difference between stress that motivates you versus stress that overwhelms you? What do you think determines that difference?`,
                cognitive: `Cognitive psychology gives us amazing insights into how our minds work! I'm curious about your thoughts - how do you think our thought patterns influence our emotions and behaviors? Can you think of an example from your own life?`,
                assessment: `Mental health assessment is crucial for effective treatment. What do you think makes a good psychological assessment? If you were trying to understand someone's mental health, what kinds of information would you want to gather?`
            };
            addMessage(responses[topic] || "Let's explore this topic together! What specific aspect interests you most?");
        }
        function handleKeyPress(event) {
            if (event.key === 'Enter') {
                sendMessage();
            }
        }
        // Feedback system: simple alert for now (can replace with backend call)
        function submitFeedback(rating) {
            document.getElementById('topicFeedback').style.display = 'none';
            if (rating === 'yes') {
                alert('Thank you for your feedback! Glad you found the topic helpful.');
            } else {
                alert('Thank you for your honesty! We aim to keep improving.');
            }
        }
        // Resource Modal controls
        function showResources() {
            document.getElementById('resourcesModal').style.display = 'block';
        }
        function closeResources() {
            document.getElementById('resourcesModal').style.display = 'none';
        }
        // Initial tutor enhancement
        setTimeout(() => {
            addMessage("I'm designed to help you think like a psychologist - I'll ask questions that encourage you to analyze, synthesize, and apply concepts rather than just memorize them. Remember, in psychology, there are often multiple valid perspectives on complex issues. What draws you to studying mental health?");
        }, 2000);
    </script>
